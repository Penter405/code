/************************************************************
 * 常用 C++ include（LeetCode / 刷題 / 比賽版本）
 * 下面每個 include 後面都有功能解釋。
 ************************************************************/
```
#include <vector>          // 動態陣列 (最常用的容器)
#include <string>          // 字串處理
#include <queue>           // BFS、隊列、priority_queue
#include <stack>           // 括號、單調堆疊、遞迴改迭代
#include <algorithm>       // sort, reverse, max, min, lower_bound
#include <unordered_map>   // Hash Map，Two Sum、計數用
#include <unordered_set>   // Hash Set，判斷是否出現過
#include <map>             // 有序 map
#include <set>             // 有序 set
#include <limits>          // INT_MAX, LLONG_MAX 等常數

using namespace std;
```
